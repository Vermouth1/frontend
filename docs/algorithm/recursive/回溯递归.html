# 回溯递归

# 试例1

电商 sku 全排列

let names = ["iPhone X", "iPhone XS"]

let colors = ["黑色", "白色"]

let storages = ["64g", "256g"]

需要把他们的所有组合穷举出来，最终得到这样一个数组：

[
  ["iPhone X", "黑色", "64g"],
  ["iPhone X", "黑色", "256g"],
  ["iPhone X", "白色", "64g"],
  ["iPhone X", "白色", "256g"],
  ["iPhone XS", "黑色", "64g"],
  ["iPhone XS", "黑色", "256g"],
  ["iPhone XS", "白色", "64g"],
  ["iPhone XS", "白色", "256g"],
]

function combine(...arr) {
    let res = []; // 结果集
    // index 控制遍历到的层数，prev记录当前的结果
    // 处理完每次的结果需要 pop() 回溯
    const loop = (index, prev) => {
        let _arr = arr[index];
        for (let i = 0; i < _arr.length;i++) {
            let item = _arr[i];
            prev.push(item);
            // 此处需要全排列，所以只有都遍历完的时候，才push到结果中
            // 如果没遍历完，递归下一层
            if (index === arr.length - 1) {
                res.push(prev.slice());
            } else {
                loop(index+1, prev);
            }
            prev.pop()
        }
    }
    loop(0, []);
    return res;
}

let names = ["iPhone X", "iPhone XS"]
let colors = ["黑色", "白色"]
let storages = ["64g", "256g"]
combine(names, colors, storages)

# 试例2

给定一个整型数组, 你的任务是找到所有该数组的递增子序列，递增子序列的长度至少是2。

输入: [4, 6, 7, 7]
输出: [[4, 6], [4, 7], [4, 6, 7], [4, 6, 7, 7], [6, 7], [6, 7, 7], [7,7], [4,7,7]]

var findSubsequences = (nums) => {
    let set = new Set(); // 去重
    let res = []; // 结果集
    const loop = (index, prev) => {
        for (let i = index; i < nums.length; i++) {
            let item = nums[i];
            // 如果是第一层或者当前处理的元素比上一层的元素大，push结果
            if (prev.length === 0 || item >= prev.slice(-1)) {
                prev.push(item);
                // 题目要求子集长度最小为2，所以 > 1 时添加结果集
                if (prev.length > 1) {
                    let str = prev.join(',');
                    if (!set.has(str)) {
                        set.add(str);
                        res.push(prev.slice());
                    }
                }
                loop(i+1, prev);
                prev.pop();
            }
        }
    }
    loop(0, []);
    return res;
}
findSubsequences([4,6,7,7])

# 试例3

电话案件对应字母的所用情况

var letterCombinations = function(digits) {
    const map = {
        '2': 'abc',
        '3': 'def',
        '4': 'ghi',
        '5': 'jkl',
        '6': 'mno',
        '7': 'pqrs',
        '8': 'tuv',
        '9': 'wxyz'
    };
    let res = [];
    const loop = (index, prev) => {
        for (let i = index; i < digits.length; i++) {
            let strs = map[digits[i]];
            for (let j = 0; j < strs.length; j++) {
                prev.push(strs[j]);
                if (prev.length === digits.length) {
                    res.push(prev.join(''))
                }
                loop(i + 1, prev);
                prev.pop();
            }
        }
    }
    loop(0, []);
    return res
};
console.log(letterCombinations('23'))
